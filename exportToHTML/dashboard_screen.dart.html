23     Future<void> _loadData() async {
24       final receiptProvider =
25           Provider.of<ReceiptProvider>(context, listen: false);
26       final today = DateTime.now();
27   
28       // Nastavení dateRange na aktuální den
29       receiptProvider.updateDateRange(
30         DateTimeRange(
31           start: DateTime(today.year, today.month, today.day),
32           end: DateTime(today.year, today.month, today.day, 23, 59, 59),
33         ),
34       );
35   
36       await receiptProvider.fetchReceipts();
37     }
38   
39     @override
40     Widget build(BuildContext context) {
41       return Scaffold(
42         appBar: AppBar(
43           title: const Text(
44             "Nástěnka",
45             style: TextStyle(color: Colors.white),
46           ),
47           backgroundColor: Colors.grey[850],
48         ),
49         body: FutureBuilder<void>(
50           future: _fetchData,
51           builder: (context, snapshot) {
52             if (snapshot.connectionState == ConnectionState.waiting) {
53               return const Center(child: CircularProgressIndicator());
54             } else if (snapshot.hasError) {
55               return Center(
56                 child: Text(
57                   'Chyba při načítání dat: ${snapshot.error}',
58                   style: const TextStyle(color: Colors.red),
59                 ),
60               );
61             } else {
62               return _buildDashboardContent(context);
63             }
64           },
65         ),
66       );
67     }

69     Widget _buildDashboardContent(BuildContext context) {
70       final receiptProvider = Provider.of<ReceiptProvider>(context);
71       final receipts = receiptProvider.receipts;
72   
73       final int todayReceiptsCount = receipts.length;
74       final double todaySales = receipts.fold(
75         0.0,
76         (sum, receipt) => sum + (receipt['total'] as num).toDouble(),
77       );
78   
79       final Map<String, Map<String, dynamic>> productData = {};
80       for (var receipt in receipts) {
81         for (var item in receipt['items']) {
82           if (!productData.containsKey(item['text'])) {
83             productData[item['text']] = {
84               'name': item['text'],
85               'quantity': 0.0,
86               'revenue': 0.0,
87             };
88           }
89           productData[item['text']]!['quantity'] +=
90               (item['quantity'] as num).toDouble();
91           productData[item['text']]!['revenue'] +=
92               (item['priceToPay'] as num).toDouble();
93         }
94       }
95   
96       final topProducts = productData.values.toList()
97         ..sort((a, b) => b['revenue'].compareTo(a['revenue']));
98       final displayedTopProducts = topProducts.take(5).toList();
99   
100      return SingleChildScrollView(
101        child: Padding(
102          padding: const EdgeInsets.all(16.0),
103          child: Column(
104            crossAxisAlignment: CrossAxisAlignment.start,
105            children: [
106              Row(
107                mainAxisAlignment: MainAxisAlignment.spaceBetween,
108                children: [
109                  _buildSummaryBlock(
110                      'Počet účtenek', todayReceiptsCount.toString()),
111                  _buildSummaryBlock(
112                      'Dnešní tržba', '${todaySales.toStringAsFixed(2)} Kč'),
113                ],
114              ),
115              const SizedBox(height: 20),
116              const Text(
117                'Nejvyšší prodeje:',
118                style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
119              ),
120              const SizedBox(height: 10),
121              _buildTopProductsTable(displayedTopProducts),
122              const SizedBox(height: 20),
123              const Text(
124                'Hodinový graf tržeb:',
125                style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
126              ),
127              const SizedBox(height: 10),
128              _buildHourlyRevenueChart(
129                  receiptProvider), // Add the bar chart widget
130              const SizedBox(height: 20),
131              const Text(
132                'Tržba podle typu platby:',
133                style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
134              ),
135              const SizedBox(height: 10),
136              _buildDynamicPieChart(receiptProvider),
137            ],
138          ),
139        ),
140      );
141    }

143    Widget _buildDynamicPieChart(ReceiptProvider receiptProvider) {
144      // Shromáždění tržeb podle typu platby
145      final paymentData = {
146        'Hotově': 0.0,
147        'Kartou': 0.0,
148        'Převodem': 0.0,
149        'Ostatní': 0.0,
150      };
151  
152      for (var receipt in receiptProvider.receipts) {
153        final paymentType = receipt['paymentType'];
154        final total = (receipt['total'] as num).toDouble();
155  
156        if (paymentType == 'CASH') {
157          paymentData['Hotově'] = paymentData['Hotově']! + total;
158        } else if (paymentType == 'CARD') {
159          paymentData['Kartou'] = paymentData['Kartou']! + total;
160        } else if (paymentType == 'BANK') {
161          paymentData['Převodem'] = paymentData['Převodem']! + total;
162        } else {
163          paymentData['Ostatní'] = paymentData['Ostatní']! + total;
164        }
165      }
166  
167      // Předdefinované barvy pro typy plateb
168      final Map<String, Color> paymentColors = {
169        'Hotově': Colors.yellow,
170        'Kartou': Colors.blue,
171        'Převodem': Colors.red,
172        'Ostatní': Colors.grey,
173      };
174  
175      // Převod dat na sekce pro koláčový graf
176      final chartSections = paymentData.entries.map((entry) {
177        final value = entry.value;
178        final percentage = value > 0
179            ? (value / receiptProvider.totalRevenue * 100).toDouble()
180            : 0.0;
181  
182        return PieChartSectionData(
183            color: paymentColors[entry.key],
184            value: percentage,
185            title: '${percentage.toStringAsFixed(1)}%',
186            radius: 80,
187            titleStyle: const TextStyle(
188              fontSize: 16,
189              fontWeight: FontWeight.bold,
190            ));
191      }).toList();
192  
193      // Vytvoření legendy
194      final legendItems = paymentData.keys.map((key) {
195        return Padding(
196          padding: const EdgeInsets.symmetric(vertical: 4.0),
197          child: Row(
198            children: [
199              Container(
200                width: 16,
201                height: 16,
202                color: paymentColors[key],
203                margin: const EdgeInsets.only(right: 6.0),
204              ),
205              Text(key, style: const TextStyle(fontSize: 14)),
206            ],
207          ),
208        );
209      }).toList();
210  
211      return Container(
212        padding: const EdgeInsets.all(8.0),
213        decoration: BoxDecoration(
214          color: Colors.white,
215          borderRadius: BorderRadius.circular(12.0),
216          border: Border.all(color: Colors.grey, width: 1),
217        ),
218        child: Row(
219          children: [
220            // Koláčový graf
221            Expanded(
222              flex: 3,
223              child: SizedBox(
224                height: 300,
225                child: PieChart(
226                  PieChartData(
227                    sections: chartSections,
228                    centerSpaceRadius: 40,
229                    sectionsSpace: 4,
230                    borderData: FlBorderData(show: false),
231                  ),
232                ),
233              ),
234            ),
235            const SizedBox(width: 8),
236            // Legenda
237            Expanded(
238              flex: 1,
239              child: Column(
240                crossAxisAlignment: CrossAxisAlignment.start,
241                children: [
242                  const Text(
243                    'Legenda:',
244                    style: TextStyle(
245                      fontSize: 16,
246                      fontWeight: FontWeight.bold,
247                    ),
248                  ),
249                  const SizedBox(height: 8),
250                  ...legendItems,
251                ],
252              ),
253            ),
254          ],
255        ),
256      );
257    }

371    Widget _buildHourlyRevenueChart(ReceiptProvider receiptProvider) {
372      // Initialize an array to hold hourly revenue
373      List<double> hourlyRevenue = List.generate(24, (index) => 0.0);
374  
375      // Aggregate revenue by hour
376      for (var receipt in receiptProvider.receipts) {
377        if (receipt['dateTime'] != null && receipt['total'] != null) {
378          try {
379            DateTime date = DateTime.parse(receipt['dateTime']);
380            double revenue = (receipt['total'] as num).toDouble();
381            hourlyRevenue[date.hour] += revenue;
382          } catch (e) {
383            print('Chyba při zpracování účtenky: $e');
384          }
385        }
386      }
387  
388      // Dynamicky určujeme rozsah hodin
389      int minHour = hourlyRevenue.indexWhere((value) => value > 0);
390      int maxHour = hourlyRevenue.lastIndexWhere((value) => value > 0);
391  
392      // Zajištění minimálního rozsahu 8:00–18:00
393      minHour = minHour == -1 ? 8 : min(8, minHour);
394      maxHour = maxHour == -1 ? 18 : max(18, maxHour);
395  
396      // Připravíme data pro graf pouze pro relevantní rozsah
397      final barGroups = List.generate(maxHour - minHour + 1, (index) {
398        int hour = minHour + index;
399        return BarChartGroupData(
400          x: hour,
401          barRods: [
402            BarChartRodData(
403              toY: hourlyRevenue[hour],
404              color: Colors.blueAccent,
405              width: 16,
406            ),
407          ],
408        );
409      });
410  
411      return Container(
412        height: 300,
413        padding: const EdgeInsets.all(16.0),
414        decoration: BoxDecoration(
415          color: Colors.white,
416          borderRadius: BorderRadius.circular(12.0),
417          border: Border.all(color: Colors.grey, width: 1),
418        ),
419        child: BarChart(
420          BarChartData(
421            barGroups: barGroups,
422            titlesData: FlTitlesData(
423              leftTitles: AxisTitles(
424                sideTitles: SideTitles(
425                  showTitles: true,
426                  getTitlesWidget: (value, meta) {
427                    // Zobrazit pouze popisy mimo nejvyšší hodnotu
428                    if (value.toInt() == hourlyRevenue.reduce(max).toInt()) {
429                      return const Text(''); // Odstraníme popis nejvyšší hodnoty
430                    }
431                    return Text(
432                      '${value.toInt()} Kč',
433                      style: const TextStyle(fontSize: 12),
434                    );
435                  },
436                  reservedSize: 40,
437                ),
438              ),
439              rightTitles: AxisTitles(sideTitles: SideTitles(showTitles: false)),
440              topTitles: AxisTitles(sideTitles: SideTitles(showTitles: false)),
441              bottomTitles: AxisTitles(
442                sideTitles: SideTitles(
443                  showTitles: true,
444                  reservedSize: 40,
445                  getTitlesWidget: (value, meta) {
446                    if (value.toInt() >= minHour && value.toInt() <= maxHour) {
447                      return Text(
448                        value.toInt().toString(),
449                        style: const TextStyle(fontSize: 10),
450                      );
451                    }
452                    return const Text('');
453                  },
454                ),
455              ),
456            ),
457            gridData: FlGridData(
458              show: true,
459              drawVerticalLine: false, // Odstranění svislých čar
460            ),
461            borderData:
462                FlBorderData(show: true, border: Border.all(color: Colors.grey)),
463          ),
464        ),
465      );
466    }
