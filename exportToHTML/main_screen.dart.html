11   class MainScreen extends StatefulWidget {
12     final Future<void> Function(String) updateApiKey;
13   
14     const MainScreen({super.key, required this.updateApiKey});
15   
16     @override
17     State<MainScreen> createState() => _MainScreenState();
18   }
19   
20   class _MainScreenState extends State<MainScreen> {
21     int _selectedIndex = 0;
22   
23     late final List<Widget> _screens;
24   
25     @override
26     void initState() {
27       super.initState();
28       _screens = [
29         const DashboardScreen(),
30         const ReceiptListScreen(),
31         const ProductListScreen(),
32         const CustomersScreen(),
33         SettingsScreen(updateApiKey: widget.updateApiKey),
34       ];
35     }
36   
37     void _onItemTapped(int index) {
38       setState(() {
39         _selectedIndex = index;
40       });
41   
42       // Refresh logika
43       if (index == 1) {
44         context.read<ReceiptProvider>().fetchReceipts();
45       } else if (index == 2) {
46         context.read<ProductProvider>().fetchProducts();
47       }
48     }
49   
50     @override
51     Widget build(BuildContext context) {
52       return Scaffold(
53         body: _screens[_selectedIndex],
54         bottomNavigationBar: BottomNavigationBar(
55           type: BottomNavigationBarType.fixed,
56           currentIndex: _selectedIndex,
57           onTap: _onItemTapped,
58           selectedItemColor: Colors.white,
59           unselectedItemColor: Colors.grey,
60           backgroundColor: Colors.grey[850],
61           items: const [
62             BottomNavigationBarItem(
63               icon: Icon(Icons.dashboard),
64               label: 'Nástěnka',
65             ),
66             BottomNavigationBarItem(
67               icon: Icon(Icons.receipt),
68               label: 'Prodeje',
69             ),
70             BottomNavigationBarItem(
71               icon: Icon(Icons.shopping_bag),
72               label: 'Produkty',
73             ),
74             BottomNavigationBarItem(
75               icon: Icon(Icons.people),
76               label: 'Zákazníci',
77             ),
78             BottomNavigationBarItem(
79               icon: Icon(Icons.settings),
80               label: 'Nastavení',
81             ),
82           ],
83         ),
84       );
85     }
86   }
