3    class StorageService {
4      // Klíč pro ukládání API klíče do SharedPreferences
5      static const String _apiKeyKey = 'api_key';
6    
7      /// Uloží API klíč do `SharedPreferences`.
8      static Future<void> saveApiKey(String apiKey) async {
9        SharedPreferences prefs = await SharedPreferences.getInstance();
10       await prefs.setString(_apiKeyKey, apiKey);
11       print('API klíč byl uložen: $apiKey');
12     }
13   
14     /// Načte API klíč z `SharedPreferences`.
15     static Future<String?> getApiKey() async {
16       SharedPreferences prefs = await SharedPreferences.getInstance();
17       String? apiKey = prefs.getString(_apiKeyKey);
18       print('Načtený API klíč: $apiKey');
19       return apiKey;
20     }
21   
22     /// Smaže API klíč z `SharedPreferences`.
23     static Future<void> clearApiKey() async {
24       SharedPreferences prefs = await SharedPreferences.getInstance();
25       await prefs.remove(_apiKeyKey);
26       print('API klíč byl smazán.');
27     }
28   }

30   class Utility{
31   
32     /// Normalizace textu odstraněním diakritiky
33     static String normalizeString(String input) {
34       const withDiacritics = 'áčďéěíňóřšťúůýžÁČĎÉĚÍŇÓŘŠŤÚŮÝŽ';
35       const withoutDiacritics = 'acdeeinorstuuyzACDEEINORSTUUYZ';
36   
37       return input.split('').map((char) {
38         final index = withDiacritics.indexOf(char);
39         return index != -1 ? withoutDiacritics[index] : char;
40       }).join();
41     }
42   }
